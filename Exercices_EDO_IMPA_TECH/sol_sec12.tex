\bigskip\noindent\textbf{SoluûÏûÈo 59.}\quad
Recordemos que, para $v=(v_1,v_2)$ e $w=(w_1,w_2)$, $\det(v,w)=v_1w_2-v_2w_1$, e que a curvatura com sinal de $\gamma=(\gamma_1,\gamma_2)$ ûˋ
\[
\kappa=\frac{\det(\gamma',\gamma'')}{\|\gamma'\|^3}
=\frac{\gamma_1'\gamma_2''-\gamma_2'\gamma_1''}{\big((\gamma_1')^2+(\gamma_2')^2\big)^{3/2}} .
\]
Esta quantidade ûˋ \emph{invariante por reparametrizaûÏûÈo} que preserva a orientaûÏûÈo: se $\gamma=\sigma\circ\varphi$ com $\varphi'>0$, a regra da cadeia mostra que numerador e denominador sûÈo ambos multiplicados por $(\varphi')^3$, que se cancela. Por isso podemos calcular $\kappa$ em qualquer parametrizaûÏûÈo conveniente.

\begin{enumerate}[label=(\alph*)]

\item Para o grûÀfico $y=f(x)$ tomamos a parametrizaûÏûÈo natural $\gamma(x)=(x,f(x))$. EntûÈo
\[
\gamma'(x)=(1,f'(x)),\qquad \gamma''(x)=(0,f''(x)),
\]
de modo que
\[
\det(\gamma',\gamma'')=\det\begin{pmatrix}1&0\\ f'&f''\end{pmatrix}=1\cdot f''-f'\cdot0=f''(x),
\qquad
\|\gamma'(x)\|=\sqrt{1+f'(x)^2}.
\]
Logo
\[
\boxed{\,\kappa(x)=\dfrac{f''(x)}{\bigl(1+f'(x)^2\bigr)^{3/2}}\,}.
\]
O sinal de $\kappa$ coincide com o de $f''$: a curva ûˋ convexa ($\kappa>0$) onde $f$ ûˋ convexa, cûÇncava onde $f$ ûˋ cûÇncava.

\item \emph{Reta.} Parametrizando $\gamma(t)=p+tv$ com $v\neq0$, temos $\gamma'=v$ constante e $\gamma''=0$, donde $\det(\gamma',\gamma'')=0$ e
\[
\boxed{\,\kappa\equiv0\,}.
\]
Geometricamente, a direûÏûÈo tangente nunca muda; curvatura nula caracteriza, de fato, os segmentos de reta (ver Problema~61(b)).

\emph{Circunferûˆncia de raio $R$.} Parametrizando $\gamma(t)=c+R(\cos t,\sin t)$,
\[
\gamma'(t)=R(-\sin t,\cos t),\qquad \gamma''(t)=R(-\cos t,-\sin t),
\]
\[
\det(\gamma',\gamma'')
=\det\begin{pmatrix}-R\sin t&-R\cos t\\ R\cos t&-R\sin t\end{pmatrix}
=R^2(\sin^2t+\cos^2t)=R^2,
\qquad \|\gamma'\|=R,
\]
logo
\[
\boxed{\,\kappa=\dfrac{R^2}{R^3}=\dfrac1R\,}.
\]
Percorrida no sentido oposto (horûÀrio) obterûÙamos $\kappa=-1/R$. Assim a curvatura de uma circunferûˆncia ûˋ constante e igual ao inverso do raio (a menos de sinal, dado pela orientaûÏûÈo): quanto menor o raio, maior a curvatura, em acordo com a intuiûÏûÈo.

\item Na espiral de Euler a curva estûÀ parametrizada por comprimento de arco e $\kappa(s)=s$. Como $\theta'(s)=\kappa(s)=s$ e podemos tomar $\theta(0)=0$, integrando,
\[
\theta(s)=\int_0^s u\,\d u=\frac{s^2}{2}.
\]
Para uma curva p.c.a.\ o vetor tangente unitûÀrio ûˋ $\gamma'(s)=(\cos\theta(s),\sin\theta(s))$ (ver Problema~60). Logo
\[
x'(s)=\cos\!\Big(\frac{s^2}{2}\Big),\qquad y'(s)=\sin\!\Big(\frac{s^2}{2}\Big),
\]
e, fixando $\gamma(0)=(0,0)$, integramos termo a termo:
\[
\boxed{\,\gamma(s)=\left(\int_0^s\cos\!\frac{u^2}{2}\,\d u,\ \int_0^s\sin\!\frac{u^2}{2}\,\d u\right)\,}.
\]
Estas sûÈo exatamente as integrais de Fresnel. Como $\|\gamma'\|=1$ por construûÏûÈo, a parametrizaûÏûÈo ûˋ de fato por comprimento de arco, e a curvatura recuperada ûˋ $\theta'(s)=s$, como desejado. Quando $s\to+\infty$ as integrais de Fresnel convergem (a $\sqrt{\pi/2}\,$, apû°s reescalonamento), e a curva espirala em torno de um ponto-limite, com curvatura crescendo sem cota; analogamente para $s\to-\infty$.

\end{enumerate}


\bigskip\noindent\textbf{SoluûÏûÈo 60.}\quad
Dada $\kappa:[0,L]\to\R$ contûÙnua, $\theta_0\in\R$ e $p_0\in\R^2$, definimos
\[
\theta(s)=\theta_0+\int_0^s\kappa(u)\,\d u,\qquad
\gamma(s)=p_0+\int_0^s\big(\cos\theta(u),\sin\theta(u)\big)\,\d u,
\]
de modo que $\gamma'(s)=(\cos\theta(s),\sin\theta(s))$ e $\gamma(0)=p_0$. Como $\kappa$ ûˋ contûÙnua, $\theta\in C^1$ e $\gamma\in C^2$.

\begin{enumerate}[label=(\alph*)]

\item Diretamente,
\[
|\gamma'(s)|=\sqrt{\cos^2\theta(s)+\sin^2\theta(s)}=1
\]
para todo $s$. Logo o comprimento de $\gamma$ entre $0$ e $s$ ûˋ $\int_0^s|\gamma'|=s$, isto ûˋ, $s$ ûˋ o parûÂmetro de comprimento de arco. \qquad$\blacksquare$

\item Sendo $\gamma$ p.c.a., $\|\gamma'\|\equiv1$ e a curvatura com sinal reduz-se a $\kappa_\gamma=\det(\gamma',\gamma'')$. Derivando $\gamma'(s)=(\cos\theta,\sin\theta)$,
\[
\gamma''(s)=\theta'(s)\,(-\sin\theta,\cos\theta).
\]
Portanto
\[
\det(\gamma',\gamma'')
=\det\begin{pmatrix}\cos\theta & -\theta'\sin\theta\\ \sin\theta & \theta'\cos\theta\end{pmatrix}
=\theta'\cos^2\theta+\theta'\sin^2\theta=\theta'(s).
\]
Como $\theta'(s)=\kappa(s)$ por construûÏûÈo, concluûÙmos que a curvatura com sinal de $\gamma$ ûˋ exatamente $\kappa(s)$. Em particular, $\gamma''=\kappa\,N$ com $N=(-\sin\theta,\cos\theta)$ o normal unitûÀrio obtido girando $T=\gamma'$ de $+\pi/2$. \qquad$\blacksquare$

\item \emph{(Teorema fundamental das curvas planas.)} Sejam $\gamma_1,\gamma_2:[0,L]\to\R^2$ duas curvas $C^2$ parametrizadas por comprimento de arco com a mesma curvatura com sinal $\kappa$. Escrevendo $T_i=\gamma_i'$ e $N_i=(-(\gamma_i')_2,(\gamma_i')_1)$, as equaûÏûçes de Frenet planas (parte (b) e Problema~61(a)) dûÈo
\[
T_i'=\kappa\,N_i,\qquad N_i'=-\kappa\,T_i,\qquad i=1,2.
\]
Apû°s aplicar a $\gamma_2$ a rotaûÏûÈo que leva $T_2(0)$ em $T_1(0)$ (uma rotaûÏûÈo aplicada a uma curva preserva a curvatura com sinal e leva o referencial de Frenet no referencial girado) e a translaûÏûÈo que leva o seu ponto inicial em $\gamma_1(0)$, podemos supor
\[
\gamma_1(0)=\gamma_2(0),\qquad T_1(0)=T_2(0),
\]
e entûÈo tambûˋm $N_1(0)=N_2(0)$, pois $N_i$ ûˋ obtido de $T_i$ por uma rotaûÏûÈo fixa de $+\pi/2$.

Considere a funûÏûÈo de energia
\[
\Phi(s):=|T_1(s)-T_2(s)|^2+|N_1(s)-N_2(s)|^2\ \ge 0,\qquad \Phi(0)=0.
\]
Derivando e usando as equaûÏûçes de Frenet,
\[
\tfrac12\Phi'
=\langle T_1-T_2,\,T_1'-T_2'\rangle+\langle N_1-N_2,\,N_1'-N_2'\rangle
=\kappa\,\langle T_1-T_2,\,N_1-N_2\rangle-\kappa\,\langle N_1-N_2,\,T_1-T_2\rangle=0.
\]
Logo $\Phi$ ûˋ constante, e como $\Phi(0)=0$ temos $\Phi\equiv0$. Em particular $T_1(s)=T_2(s)$ para todo $s$. Integrando,
\[
\gamma_1(s)-\gamma_2(s)=\gamma_1(0)-\gamma_2(0)+\int_0^s\big(T_1-T_2\big)=0.
\]
Portanto $\gamma_1\equiv\gamma_2$: as duas curvas coincidem. Assim a curvatura com sinal determina a curva a menos de um movimento rûÙgido direto, e a parte (a)--(b) mostra que toda funûÏûÈo contûÙnua $\kappa$ ûˋ, de fato, a curvatura de alguma curva. \qquad$\blacksquare$

\end{enumerate}


\bigskip\noindent\textbf{SoluûÏûÈo 61.}\quad
Seja $\gamma:[0,L]\to\R^2$ de classe $C^2$, parametrizada por comprimento de arco ($\|\gamma'\|\equiv1$), com curvatura com sinal constante $\kappa\equiv k$. Pomos
\[
T(s)=\gamma'(s),\qquad N(s)=(-T_2(s),T_1(s)),
\]
de modo que $\{T,N\}$ ûˋ, em cada $s$, uma base ortonormal positivamente orientada ($N$ ûˋ $T$ girado de $+\pi/2$).

\begin{enumerate}[label=(\alph*)]

\item Como $\|T\|^2=T_1^2+T_2^2\equiv1$, derivando obtemos $\langle T,T'\rangle=0$: o vetor $T'$ ûˋ ortogonal a $T$, logo ûˋ mû¤ltiplo de $N$. Escreva $T'=\lambda N$ para alguma funûÏûÈo $\lambda$. EntûÈo
\[
\det(T,T')=\det(T,\lambda N)=\lambda\,\det(T,N)=\lambda\big(T_1\cdot T_1-T_2\cdot(-T_2)\big)=\lambda(T_1^2+T_2^2)=\lambda.
\]
Por outro lado, sendo $\gamma$ p.c.a., $\det(T,T')=\det(\gamma',\gamma'')=\kappa=k$. Logo $\lambda=k$ e
\[
\boxed{\,T'(s)=k\,N(s)\,}.
\]
Para $N$, derivamos $N=(-T_2,T_1)$:
\[
N'=(-T_2',T_1')=-(T_1',T_2')\ \text{girado}\ \ \Longrightarrow\ \
N'=(-(kN)_2,(kN)_1).
\]
Como $N=(-T_2,T_1)$, temos $N_1=-T_2$, $N_2=T_1$, e usando $T'=kN=k(-T_2,T_1)$,
\[
N'=(-T_2',\,T_1')=(-kT_1,\,-kT_2)=-k\,(T_1,T_2)=-k\,T.
\]
Portanto
\[
\boxed{\,N'(s)=-k\,T(s)\,}.
\]
Estas sûÈo as equaûÏûçes de Frenet planas no caso de curvatura constante. \qquad$\blacksquare$

\item Se $k=0$, entûÈo $T'\equiv0$ por (a), logo $T(s)=T(0)=:v$ ûˋ um vetor constante unitûÀrio. Integrando $\gamma'=v$,
\[
\gamma(s)=\gamma(0)+s\,v,
\]
que ûˋ a parametrizaûÏûÈo de uma reta. Assim a imagem de $\gamma$ estûÀ contida na reta $\{\gamma(0)+t\,v:t\in\R\}$. \qquad$\blacksquare$

\item Suponha $k\neq0$ e defina o ``centro''
\[
c(s):=\gamma(s)+\frac1k\,N(s).
\]
Derivando e usando (a),
\[
c'(s)=\gamma'(s)+\frac1k\,N'(s)=T(s)+\frac1k\,(-k\,T(s))=T(s)-T(s)=0.
\]
Logo $c(s)\equiv c$ ûˋ constante. DaûÙ
\[
\gamma(s)-c=-\frac1k\,N(s)\quad\Longrightarrow\quad
\|\gamma(s)-c\|=\frac1{|k|}\,\|N(s)\|=\frac1{|k|},
\]
pois $\|N\|\equiv1$. Portanto todo ponto da imagem de $\gamma$ dista exatamente $1/|k|$ do ponto fixo $c$: a imagem estûÀ contida na circunferûˆncia de centro $c$ e raio $1/|k|$. Isto recupera, em sentido inverso, o cûÀlculo do Problema~59(b): curvatura com sinal constante nûÈo nula caracteriza os arcos de circunferûˆncia. \qquad$\blacksquare$

\end{enumerate}


\bigskip\noindent\textbf{SoluûÏûÈo 62.}\quad
Seja $\gamma:[0,L]\to\R^2$ suave, simples, fechada e estritamente convexa, parametrizada por comprimento de arco, com curvatura com sinal $\kappa>0$ em todo ponto. Como $\|\gamma'\|\equiv1$, podemos escrever
\[
\gamma'(s)=(\cos\theta(s),\sin\theta(s)),
\]
onde $\theta$ ûˋ uma escolha \emph{contûÙnua} (de fato $C^1$) do ûÂngulo tangente, com $\theta'(s)=\kappa(s)$ (Problema~60(b)). A funûÏûÈo $s\mapsto\gamma'(s)$ ûˋ a aplicaûÏûÈo tangente; identificando vetores unitûÀrios com pontos do cûÙrculo $S^1$, ela ûˋ a chamada \emph{aplicaûÏûÈo de Gauss} ou \emph{indicatriz tangente}, e $\theta$ ûˋ um seu levantamento contûÙnuo a $\R$.

\begin{enumerate}[label=(\alph*)]

\item Como $\gamma$ ûˋ fechada e suave (de classe $C^1$ com extremidades coladas suavemente), $\gamma'(L)=\gamma'(0)$, isto ûˋ, os vetores tangentes inicial e final coincidem. Para um levantamento contûÙnuo do ûÂngulo isto forûÏa
\[
\theta(L)-\theta(0)=2\pi m
\]
para algum inteiro $m\in\mathbb{Z}$ --- o \emph{ûÙndice de rotaûÏûÈo} (ou nû¤mero de voltas) da curva. O teorema do ûÙndice de rotaûÏûÈo de Hopf (\emph{Umlaufsatz}) afirma que, para uma curva fechada \emph{simples} (sem auto-interseûÏûçes), $m=\pm1$, o sinal dependendo da orientaûÏûÈo de percurso.

No nosso caso o sinal ûˋ determinado pela convexidade: como $\theta'(s)=\kappa(s)>0$ para todo $s$, a funûÏûÈo $\theta$ ûˋ \emph{estritamente crescente}, logo $\theta(L)-\theta(0)>0$, forûÏando $m\ge1$. Pelo Umlaufsatz $m=1$. ConcluûÙmos que, ao percorrer a curva uma vez, a direûÏûÈo tangente dûÀ exatamente uma volta completa:
\[
\boxed{\,\theta(L)=\theta(0)+2\pi\,}.
\]
Intuitivamente: numa curva convexa fechada percorrida no sentido positivo, a tangente gira sempre no mesmo sentido (pois $\kappa>0$) e, por ser a curva simples e fechada, perfaz precisamente uma û¤nica revoluûÏûÈo. \qquad$\blacksquare$

\item Integrando a relaûÏûÈo intrûÙnseca $\theta'=\kappa$ sobre $[0,L]$ e usando (a),
\[
\int_0^L\kappa(s)\,\d s=\int_0^L\theta'(s)\,\d s=\theta(L)-\theta(0)=2\pi.
\]
Logo
\[
\boxed{\,\int_0^L\kappa(s)\,\d s=2\pi\,}.
\]
Esta ûˋ a forma elementar do teorema de Gauss--Bonnet no plano: a curvatura total de uma curva convexa simples fechada ûˋ $2\pi$. \qquad$\blacksquare$

\item A convexidade estrita equivale a $\kappa>0$ em todo ponto, o que torna $\theta$ \emph{estritamente monû°tona crescente}. Suponha, por absurdo, que a tangente desse $m\ge2$ voltas, isto ûˋ, $\theta(L)-\theta(0)=2\pi m\ge4\pi$. Por monotonia estrita e pelo teorema do valor intermediûÀrio, $\theta$ assumiria \emph{cada} valor angular (mû°dulo $2\pi$) em pelo menos $m\ge2$ instantes distintos $s_1<s_2$ em $[0,L)$; nesses dois instantes os vetores tangentes $\gamma'(s_1),\gamma'(s_2)$ seriam paralelos e \emph{de mesmo sentido}. Para uma curva convexa simples isso ûˋ impossûÙvel com $s_1\neq s_2$ no interior do perûÙodo: o teorema do ûÙndice de rotaûÏûÈo garante que, sendo a curva simples, o levantamento contûÙnuo varre o intervalo $[\theta(0),\theta(0)+2\pi)$ uma û¤nica vez, cada direûÏûÈo tangente positivamente orientada ocorrendo em um û¤nico instante (geometricamente, uma curva convexa admite, em cada direûÏûÈo, exatamente uma reta tangente com aquele sentido). Voltas adicionais exigiriam que a curva se cruzasse ou perdesse a convexidade ($\kappa$ mudaria de sinal para ``desfazer'' parte da rotaûÏûÈo), contradizendo as hipû°teses. Portanto $m=1$, sem voltas adicionais. \qquad$\blacksquare$

\end{enumerate}


\bigskip\noindent\textbf{SoluûÏûÈo 63.}\quad
Sejam $a>0$, $L>2a$. Procuramos maximizar a ûÀrea
\[
A[y]=\int_{-a}^a y(x)\,\d x
\]
entre as funûÏûçes suaves $y:[-a,a]\to[0,\infty)$ com $y(\pm a)=0$ e \emph{comprimento prescrito}
\[
\mathcal L[y]=\int_{-a}^a\sqrt{1+y'(x)^2}\,\d x=L.
\]
A condiûÏûÈo $L>2a$ garante que o vûÙnculo ûˋ compatûÙvel com grûÀficos nûÈo triviais (o segmento reto $y\equiv0$ tem comprimento $2a<L$). Suponhamos, conforme o enunciado, que exista um maximizador $y$, suave e positivo em $(-a,a)$.

\medskip
\noindent\emph{VariaûÏûÈo com vûÙnculo (multiplicador de Lagrange).}
Como $\mathcal L[y]=L$ ûˋ uma restriûÏûÈo integral (isoperimûˋtrica), o mûˋtodo dos multiplicadores de Lagrange para o cûÀlculo das variaûÏûçes afirma: existe uma constante $\lambda\in\R$ tal que $y$ ûˋ ponto crûÙtico do funcional aumentado
\[
J[y]:=A[y]-\lambda\,\mathcal L[y]=\int_{-a}^a\Big(y-\lambda\sqrt{1+y'^2}\Big)\,\d x=:\int_{-a}^a F(y,y')\,\d x,
\]
\emph{sem} vûÙnculo, entre as funûÏûçes com $y(\pm a)=0$. (Justificativa: para qualquer par de perturbaûÏûçes admissûÙveis $\eta_1,\eta_2$ que se anulam em $\pm a$, considere $y+t_1\eta_1+t_2\eta_2$; a condiûÏûÈo de extremo de $A$ sujeito a $\mathcal L=L$ na vizinhanûÏa de $y$, supondo $\delta\mathcal L\not\equiv0$, equivale, pelo teorema dos multiplicadores em dimensûÈo finita aplicado a $(t_1,t_2)\mapsto(A,\mathcal L)$, û  existûˆncia de $\lambda$ com $\delta A=\lambda\,\delta\mathcal L$.)

\medskip
\noindent\emph{EquaûÏûÈo de Euler--Lagrange.}
Para $F(y,y')=y-\lambda\sqrt{1+y'^2}$ temos
\[
F_y=1,\qquad F_{y'}=-\lambda\,\frac{y'}{\sqrt{1+y'^2}}.
\]
A equaûÏûÈo de Euler--Lagrange $\dfrac{\d}{\d x}F_{y'}=F_y$ fica
\begin{equation}\label{eq:63EL}
-\lambda\,\frac{\d}{\d x}\!\left(\frac{y'}{\sqrt{1+y'^2}}\right)=1.
\end{equation}
Calculando a derivada,
\[
\frac{\d}{\d x}\!\left(\frac{y'}{\sqrt{1+y'^2}}\right)
=\frac{y''\sqrt{1+y'^2}-y'\cdot\dfrac{y'y''}{\sqrt{1+y'^2}}}{1+y'^2}
=\frac{y''(1+y'^2)-y'^2y''}{(1+y'^2)^{3/2}}
=\frac{y''}{(1+y'^2)^{3/2}}.
\]
Reconhecemos a curvatura com sinal do grûÀfico,
\[
\kappa=\frac{y''}{(1+y'^2)^{3/2}}.
\]
Logo \eqref{eq:63EL} torna-se
\[
-\lambda\,\kappa(x)=1\qquad\Longrightarrow\qquad
\boxed{\,\kappa(x)=-\frac1\lambda=\text{const}\,}.
\]
(O multiplicador $\lambda\neq0$, pois caso contrûÀrio \eqref{eq:63EL} daria $0=1$.) Portanto o maximizador tem \emph{curvatura com sinal constante}.

\medskip
\noindent\emph{ConclusûÈo: arco de circunferûˆncia.}
Pelo Problema~61(c), uma curva $C^2$ com curvatura com sinal constante $\kappa_0\neq0$ tem imagem contida numa circunferûˆncia de raio $1/|\kappa_0|=|\lambda|$. (NûÈo pode ser $\kappa_0=0$: uma reta tem comprimento $2a$ entre $x=-a$ e $x=a$, abaixo de $L$; equivalentemente, $\kappa\equiv0$ daria $y$ afim, e com $y(\pm a)=0$ seria $y\equiv0$, que nûÈo maximiza a ûÀrea e viola o comprimento.) Assim o grûÀfico de $y$ ûˋ um \emph{arco de circunferûˆncia} de raio $R=|\lambda|$, simûˋtrico em relaûÏûÈo ao eixo $x=0$ (a simetria do problema sob $x\mapsto-x$ e a unicidade do arco com extremos $(\pm a,0)$, comprimento $L$ e altura positiva fixam o arco como o segmento circular sobre a corda $[-a,a]$). Em suma, dentre todos os grûÀficos de comprimento $L$ ligando $(-a,0)$ a $(a,0)$, o de maior ûÀrea ûˋ o arco circular --- a versûÈo ``de grûÀfico'' da desigualdade isoperimûˋtrica. \qquad$\blacksquare$


\bigskip\noindent\textbf{SoluûÏûÈo 64.}\quad
A curva estûÀ parametrizada por comprimento de arco e
\[
\theta'(s)=\kappa(s),\qquad x'(s)=\cos\theta(s),\qquad y'(s)=\sin\theta(s),\qquad s\ge0,
\]
com $\kappa$ contûÙnua em $[0,\infty)$ e $\displaystyle\int_0^\infty|\kappa(s)|\,\d s<\infty$. Em particular $\kappa$ ûˋ absolutamente integrûÀvel em $[0,\infty)$.

\begin{enumerate}[label=(\alph*)]

\item Para $0\le s\le t$,
\[
\theta(t)-\theta(s)=\int_s^t\kappa(u)\,\d u,\qquad
|\theta(t)-\theta(s)|\le\int_s^t|\kappa(u)|\,\d u\le\int_s^\infty|\kappa(u)|\,\d u.
\]
Como $\int_0^\infty|\kappa|<\infty$, a cauda $\int_s^\infty|\kappa|\to0$ quando $s\to\infty$. Logo, dado $\varepsilon>0$, existe $S$ tal que $\int_S^\infty|\kappa|<\varepsilon$, e entûÈo $|\theta(t)-\theta(s)|<\varepsilon$ para todos $t\ge s\ge S$: $\theta$ satisfaz o critûˋrio de Cauchy em $+\infty$. Portanto existe o limite finito
\[
\theta_\infty:=\lim_{s\to\infty}\theta(s)\in\R,
\]
e fazendo $t\to\infty$ na desigualdade acima obtemos a estimativa de cauda
\begin{equation}\label{eq:64tail}
|\theta(s)-\theta_\infty|\le\int_s^\infty|\kappa(u)|\,\d u\qquad\text{para todo }s\ge0.
\end{equation}
(De fato $\theta(s)-\theta_\infty=-\int_s^\infty\kappa(u)\,\d u$.) \qquad$\blacksquare$

\item O vetor tangente unitûÀrio ûˋ $T(s)=(x'(s),y'(s))=(\cos\theta(s),\sin\theta(s))$. Como $s\mapsto(\cos,\sin)$ ûˋ contûÙnua e $\theta(s)\to\theta_\infty$ por (a), segue da continuidade que
\[

\]
um vetor unitûÀrio fixo. Assim a curva possui uma \emph{direûÏûÈo assintû°tica} $T_\infty$. \qquad$\blacksquare$

\item Suponha adicionalmente $\displaystyle\int_0^\infty s|\kappa(s)|\,\d s<\infty$. Por \eqref{eq:64tail} e o teorema de Tonelli (integrando $|\kappa|\ge0$, a troca de ordem ûˋ lûÙcita),
\[
\int_0^\infty|\theta(s)-\theta_\infty|\,\d s
\le\int_0^\infty\!\left(\int_s^\infty|\kappa(u)|\,\d u\right)\d s
=\int_0^\infty\!\!\int_0^\infty\mathbf 1_{\{u>s\}}\,|\kappa(u)|\,\d s\,\d u .
\]
Para cada $u$ fixo, $\int_0^\infty\mathbf 1_{\{u>s\}}\,\d s=\int_0^u\d s=u$. Logo
\[
\int_0^\infty|\theta(s)-\theta_\infty|\,\d s\le\int_0^\infty u\,|\kappa(u)|\,\d u<\infty.
\]
Portanto
\[
\boxed{\,\int_0^\infty|\theta(s)-\theta_\infty|\,\d s\le\int_0^\infty s\,|\kappa(s)|\,\d s<\infty\,}.
\]
\qquad$\blacksquare$
